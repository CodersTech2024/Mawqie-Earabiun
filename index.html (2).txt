<script>
    let isArabic = true;

    function toggleLanguage() {
        isArabic = !isArabic;

        // Toggle visibility of Arabic and English elements
        const elementsToShow = isArabic ? 
            ['ar'] : ['en'];
        const elementsToHide = isArabic ? 
            ['en'] : ['ar'];

        elementsToShow.forEach(lang => {
            document.querySelectorAll(`[id$='-${lang}']`).forEach(el => {
                el.style.display = 'block';
            });
        });
        
        elementsToHide.forEach(lang => {
            document.querySelectorAll(`[id$='-${lang}']`).forEach(el => {
                el.style.display = 'none';
            });
        });

        // Change button text
        document.querySelector('.lang-button').textContent = isArabic ? 'اللغة الإنجليزية' : 'Arabic Language';
    }
</script>
